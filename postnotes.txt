1. initiate

touch .gitignore
npm init -y
npm install express pg sequelize
npm install -g nodemon
npm install -D sequelize-cli
npx sequelize init
npm install cors
npm install axios
npm install bcrypt
npm install jsonwebtoken
npm i -D dotenv

2. database

npx sequelize-cli db:create

npx sequelize-cli model:create --name User --attributes name:string,email:string,password:string,age:integer,gender:string,photo:string,location:string,about:string
npx sequelize-cli model:create --name Interest --attributes name:string
npx sequelize-cli model:create --name UserInterest --attributes name:string,userId:integer
npx sequelize-cli model:create --name Image --attributes imgUrl:string
npx sequelize-cli model:create --name Like --attributes authorId:integer,authorStatus:boolean,targetId_2:integer,targetStatus:boolean
npx sequelize-cli model:create --name Chat --attributes likeId:integer,message:string

npx sequelize-cli db:migrate
